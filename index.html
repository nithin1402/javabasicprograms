5.Swapping of two numbers

class Hello
{
public static void main(String[] args)
    {
       int temp;
       int a=1;
       int b=2;
       System.out.println("Before swapping");
       System.out.println("value of a="+a);
       System.out.println("value of b="+b);
       temp=a;
       a=b;
       b=temp;
       System.out.println("After Swapping");
       System.out.println("value of a="+a);
       System.out.println("value of b="+b);
}
}
